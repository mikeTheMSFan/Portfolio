                    //Get values from the user interface
                    function getValues() {
                      const payValue = parseInt(document.getElementById('payVal').value);
                      const dayValue = parseInt(document.getElementById('dayVal').value);
                      let pdArr = [];
                    
                      if (Number.isInteger(payValue) && Number.isInteger(dayValue)) {
                        //Checks if both values are number between 1 and 10
                        if (payValue > 0 && payValue <= 10 && dayValue > 0 && dayValue <= 10) {
                          pdArr = generateNumbersC(payValue, dayValue);
                        } else {
                          alert("Please choose a number between 1 and 10.");
                        }
                        displayPayDay(pdArr);
                      } else {
                        alert('Please enter an integer');
                      }
                    }
                    
                    //Checks to see if i is divisible by payVal and dayVal and pushes the
                    //string "PAYDAY" to the return array if true.
                    
                    //Checks to see if i is divisible by payVal and pushes the
                    //string "PAY" it to the return array if true.
                    
                    //Checks to see if i is divisible by dayVal and pushrs the string
                    //"DAY" it to the return array if true.
                    
                    //If all checks fail, push the integer i to the array.
                    
                    function generateNumbers(payVal, dayVal) {
                      const returnArr = [];
                      for (let i = 1; i <= 100; i++) {
                        if (i % payVal === 0 && i % dayVal === 0) {
                          returnArr.push('PAYDAY');
                        } else if (i % payVal === 0) {
                          returnArr.push('PAY');
                        } else if (i % dayVal === 0) {
                          returnArr.push('DAY');
                        } else {
                          returnArr.push(i);
                        }
                      }
                      return returnArr;
                    }
                    
                    // Case Select Version (Cleaner look)
                    function generateNumbersB(payVal, dayVal) {
                      let returnArr = [];
                      let pay = false;
                      let day = false;
                    
                      for (let i = 1; i <= 100; i++) {
                        pay = i % payVal == 0;
                        day = i % dayVal == 0;
                    
                        switch (true) {
                          case pay && day: {
                            returnArr.push('PAYDAY');
                            break;
                          }
                          case pay: {
                            returnArr.push('PAY');
                            break;
                          }
                          case day: {
                            returnArr.push('DAY');
                            break;
                          }
                          default: {
                            returnArr.push(i);
                            break;
                          }
                        }
                      }
                      return returnArr;
                    }
                    
                    //Shortwork Version (The one to rule them all)
                    function generateNumbersC(payVal, dayVal) {
                      let returnArr = [];
                    
                      for (let i = 1; i <= 100; i++) {
                        let value =
                          (i % payVal === 0 ? 'PAY' : '') + (i % dayVal === 0 ? 'DAY' : '') || i;
                        returnArr.push(value);
                      }
                      return returnArr;
                    }
                    
                    //Loop over the table and create a table row for each item
                    function displayPayDay(pdArr) {
                      let template = '';
                    
                      //get the table body element
                      const tableBody = document.getElementById('results');
                      //get template
                      const templateRow = document.getElementById('pdTemplate');
                    
                      //reset table
                      tableBody.innerHTML = '';
                    
                      for (let i = 0; i < pdArr.length; i += 5) {
                        //Creates a fragment or copy of template we made in HTML
                        let tableRow = document.importNode(templateRow.content, true);
                    
                        //Gets just the td elements and puts them into an array
                        let rowCols = tableRow.querySelectorAll('td');
                    
                        //Adds a class name based on the content of
                        //the array element.
                    
                        //Fills 'td' array with data from the passed
                        //in array.
                        rowCols[0].classList.add(pdArr[i]);
                        rowCols[0].textContent = pdArr[i];
                    
                        rowCols[1].classList.add(pdArr[i + 1]);
                        rowCols[1].textContent = pdArr[i + 1];
                    
                        rowCols[2].classList.add(pdArr[i + 2]);
                        rowCols[2].textContent = pdArr[i + 2];
                    
                        rowCols[3].classList.add(pdArr[i + 3]);
                        rowCols[3].textContent = pdArr[i + 3];
                    
                        rowCols[4].classList.add(pdArr[i + 4]);
                        rowCols[4].textContent = pdArr[i + 4];
                    
                        //Appends the filled in 'td' rows to the
                        //HTML table.
                        tableBody.appendChild(tableRow);
                      }
                    }

Pay Day App

getValues Function

This function gets the pay value and the day value from the UI. Once we revieve these values. we attempt to convert them into integers and check if the process was sucessful. If so, we check the range of the number and, if appropreate, we pass the values to the generateNumbers function. If not, we let the user know why.

generateNumbers Function

This function uses a for loop to iterate from 1 to 100, and checks to see if the current number is divisible by the pay or day value. First, we declare the return array to hold the values parsed by the for loop. Then we check to see if ‘i’ is divisible by the pay value, when true we push the string ‘PAY’ to the array. When the number is divisible by the day value, we push the string ‘DAY’ to the array. If the pay and day value are divisible by the number, we push the string ‘PAYDAY’ to the array. Otherwise, we push ‘i’ to the array. Once we push all numbers to the array, we return it to the caller.

generateNumbersB Function

This function shows a cleaner way to a achieve the same result as the generateNumbers function. First, we declare our returnArr, and we create pay and day boolean variables. As before, we iterate the numbers 1 to 100 using a for loop. When either the pay value or day value is divisible by the current number, we set pay and/or day to true. Otherwise we set them to false. We use a switch statement with the condition of true to check if pay and/or day is true. If so, we push either ‘PAY’, ‘DAY’ or ‘PAYDAY.’ Otherwise, we push ‘i’ to the array. When finished, we return the array to the caller.

generateNumbersC Function

This function shows a concise way to achieve the same result as generateNumbers and generateNumbersB. Once again, we declare a return array and we set up a for loop to iterate from 1 to 100. Inside of the loop, we use a ternary operator. The first operator checks if the pay value is divisible by ‘i’ if so, we set the value to the string ‘PAY’. If not, we set an empty string (basically nothing). Then we check the day value to see if it’s divisible by the current number. When true, we set the string ‘DAY’ to value. When false, we set an empty string. If all checks fail, we set value to ‘i.’ When we set a string, we concatenate it. We either end up with (‘PAY’ + ‘’), which equals PAY, (‘DAY’ + ‘’) which equals DAY, or (‘PAY’ + ‘DAY’), which equals PAYDAY. Once we parse each number, we push it to the array. When finished, we return the array to the caller.

displayPayDay Function

This function passes in the array formed in the generateNumbers function, formats it, and displays it to the user. First, we declare a template variable to hold HTML that we will display. We then declare a variable that allows us to use the template we’ve created. Before entering the loop, we clear the table of any stale data. In the loop, we create a fragment or a copy of the template we’ve made. We then take that fragment and put the td elements into an array named rolCols. Since the loop takes steps in multiples of five, we fill in the td array in multiples of five. Once we add all values to the td array, we pass the completed table row to the table body as a child element, and show the completed table to the user.
